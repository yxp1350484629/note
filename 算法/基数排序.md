# 基数排序

```python
def RadixSort(alist):
    i=0
    n=0
    a=max(alist)
    #根据列表中的最大值，确定最多的位数
    while a>=10**n:
        n+=1
    #有多少位，就要进行多少次整体的比较
    while i<=n:
        dict1={}
        #这个就是依次将数字放在某种容器中，然后依次取出，就得到了顺序
        for k in range(10):
            dict1.setdefault(k,[])
        #我们将位数数字根据值放在0-9为键的10个列表中，到时候依次取出，就能得到顺序
        for r in alist:
            #注意这个式子就是可以根据i的变化求出所有位数的值
            p=int(r/(10**i)%10)
            dict1[p].append(r)
        j=0
        #依次取出，字典中的列表，并直接赋值给alist，改变传入对象alist，下一次位数对比循环在改变对象alist基础上进行，
        for k in range(10):
            #如果列表中没有对象，就不比较，减少复杂度
            if dict1[k]:
                for t in dict1[k]:
                    alist[j]=t
                    j+=1
        i+=1
```
